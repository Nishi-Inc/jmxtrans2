23  package org.jmxtrans.monitoring;
24  
25  import com.google.common.base.MoreObjects;
26  import org.apache.commons.pool.impl.GenericKeyedObjectPool;
27  
28  import javax.management.MalformedObjectNameException;
29  import javax.management.ObjectName;
30  
31  /**
32   * The Class ManagedGenericKeyedObjectPool.
33   *
34   * @author marcos.lois
35   */
36  public class ManagedGenericKeyedObjectPool implements ManagedGenericKeyedObjectPoolMBean, ManagedObject {
37  
38  	/** The object name. */
39  	private ObjectName objectName;
40  
41  	/** The default pool name. */
42  	private final String poolName;
43  
44  	/** The pool. */
45  	private final GenericKeyedObjectPool pool;
46  
47  	/**
48  	 * The Constructor.
49  	 *
50  	 * @param pool     the pool
51  	 * @param poolName the pool name
52  	 */
53  	public ManagedGenericKeyedObjectPool(GenericKeyedObjectPool pool, String poolName) {
54  		this.poolName = MoreObjects.firstNonNull(poolName, "Noname");
55  		this.pool = pool;
56  	}
57  
58  	/**
59  	 * Gets the pool name.
60  	 *
61  	 * @return the pool name
62  	 */
63  	public String getPoolName() {
64  		return poolName;
65  	}
66  
67  	@Override
68  	public ObjectName getObjectName() throws MalformedObjectNameException {
69  		if (objectName == null) {
70  			objectName = new ObjectName("org.jmxtrans:Type=GenericKeyedObjectPool,PoolName=" + this.poolName + ",Name=" + this.getClass().getSimpleName() + "@" + this.hashCode());
71  		}
72  		return objectName;
73  	}
74  
75  	@Override
76  	public void setObjectName(ObjectName objectName) throws MalformedObjectNameException {
77  		this.objectName = objectName;
78  	}
79  
80  	@Override
81  	public void setObjectName(String objectName) throws MalformedObjectNameException {
82  		this.objectName = ObjectName.getInstance(objectName);
83  	}
84  
85  	@Override
86  	public int getMaxActive() {
87  		return pool.getMaxActive();
88  	}
89  
90  	@Override
91  	public int getMaxIdle() {
92  		return pool.getMaxIdle();
93  	}
94  
95  	@Override
96  	public long getMaxWait() {
97  		return pool.getMaxWait();
98  	}
99  
100 	@Override
101 	public int getMinIdle() {
102 		return pool.getMinIdle();
103 	}
104 
105 	@Override
106 	public int getNumActive() {
107 		return pool.getNumActive();
108 	}
109 
110 	@Override
111 	public int getNumIdle() {
112 		return pool.getNumIdle();
113 	}
114 
115 	@Override
116 	public void setMaxActive(int maxActive) {
117 		this.pool.setMaxActive(maxActive);
118 	}
119 
120 	@Override
121 	public void setMaxIdle(int maxIdle) {
122 		this.pool.setMaxIdle(maxIdle);
123 	}
124 
125 	@Override
126 	public void setMinIdle(int maxIdle) {
127 		this.pool.setMinIdle(maxIdle);
128 	}
129 
130 	@Override
131 	public void setMaxWait(long maxWait) {
132 		this.pool.setMaxWait(maxWait);
133 	}
134 }
