48  /**
49   * JSON Configuration parser to create {@link org.jmxtrans.embedded.EmbeddedJmxTrans}.
50   *
51   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
52   */
53  public class ConfigurationParser {
54  
55      private final Logger logger = LoggerFactory.getLogger(getClass());
56  
57      private final ObjectMapper mapper;
58  
59      {
60          mapper = new ObjectMapper();
61          mapper.setNodeFactory(new PlaceholderEnabledJsonNodeFactory());
62      }
63  
64      public Config loadConfigurations(List<String> configurationUrls) {
65          Config config = new Config();
66          for (String configurationUrl : configurationUrls) {
67              try (InputStream in = openResource(configurationUrl)) {
68                  loadConfiguration(in, config);
69              } catch (Exception e) {
70                  throw new EmbeddedJmxTransException("Exception loading configuration'" + configurationUrl + "'", e);
71              }
72          }
73          return config;
74      }
75  
76      private InputStream openResource(String configurationUrl) throws IOException {
77          if (configurationUrl.startsWith("classpath:")) {
78              logger.debug("mergeEmbeddedJmxTransConfiguration({})", configurationUrl);
79              String path = configurationUrl.substring("classpath:".length());
80  
81              return Objects.requireNonNull(Thread.currentThread().getContextClassLoader().getResourceAsStream(path), "No file found for '" + configurationUrl + "'");
82          } else {
83              return new URL(configurationUrl).openStream();
84          }
85      }
86  
87      public void loadConfiguration(InputStream configuration, Config config) throws IOException {
88  
89          JsonNode configurationRootNode = mapper.readValue(configuration, JsonNode.class);
90          for (JsonNode queryNode : configurationRootNode.path("queries")) {
91  
92              String objectName = queryNode.path("objectName").asText();
93              Query query = new Query(objectName);
94              config.addQuery(query);
95              JsonNode resultAliasNode = queryNode.path("resultAlias");
96              if (resultAliasNode.isMissingNode()) {
97              } else if (resultAliasNode.isValueNode()) {
98                  query.setResultAlias(resultAliasNode.asText());
99              } else {
100                 logger.warn("Ignore invalid node {}", resultAliasNode);
101             }
102 
103             JsonNode attributesNode = queryNode.path("attributes");
104             if (attributesNode.isMissingNode()) {
105             } else if (attributesNode.isArray()) {
106                 Iterator<JsonNode> itAttributeNode = attributesNode.elements();
107                 while (itAttributeNode.hasNext()) {
108                     JsonNode attributeNode = itAttributeNode.next();
109                     parseQueryAttributeNode(query, attributeNode);
110                 }
111             } else {
112                 logger.warn("Ignore invalid node {}", resultAliasNode);
113             }
114 
115             JsonNode attributeNode = queryNode.path("attribute");
116             parseQueryAttributeNode(query, attributeNode);
117             logger.trace("Add {}", query);
118         }
119 
120         List<OutputWriter> outputWriters = parseOutputWritersNode(configurationRootNode);
121         config.addOutputWriters(outputWriters);
122 
123         JsonNode queryIntervalInSecondsNode = configurationRootNode.path("queryIntervalInSeconds");
124         if (!queryIntervalInSecondsNode.isMissingNode()) {
125             config.setQueryIntervalInSeconds(queryIntervalInSecondsNode.asInt());
126         }
127 
128         JsonNode exportBatchSizeNode = configurationRootNode.path("exportBatchSize");
129         if (!exportBatchSizeNode.isMissingNode()) {
130             config.setExportedBatchSize(exportBatchSizeNode.asInt());
131         }
132 
133         JsonNode numQueryThreadsNode = configurationRootNode.path("numQueryThreads");
134         if (!numQueryThreadsNode.isMissingNode()) {
135             config.setNumQueryThreads(numQueryThreadsNode.asInt());
136         }
137 
138         JsonNode exportIntervalInSecondsNode = configurationRootNode.path("exportIntervalInSeconds");
139         if (!exportIntervalInSecondsNode.isMissingNode()) {
140             config.setExportIntervalInSeconds(exportIntervalInSecondsNode.asInt());
141         }
142 
143         JsonNode numExportThreadsNode = configurationRootNode.path("numExportThreads");
144         if (!numExportThreadsNode.isMissingNode()) {
145             config.setNumExportThreads(numExportThreadsNode.asInt());
146         }
147     }
148 
149     private List<OutputWriter> parseOutputWritersNode(@Nonnull JsonNode outputWritersParentNode) {
150         JsonNode outputWritersNode = outputWritersParentNode.path("outputWriters");
151         List<OutputWriter> outputWriters = new ArrayList<OutputWriter>();
152         if (outputWritersNode.isMissingNode()) {
153         } else if (outputWritersNode.isArray()) {
154             for (JsonNode outputWriterNode : outputWritersNode) {
155                 try {
156                     String className = outputWriterNode.path("@class").asText();
157                     OutputWriter outputWriter = instantiateOutputWriter(outputWriterNode, className);
158                     logger.trace("Add {}", outputWriter);
159                     outputWriters.add(outputWriter);
160                 } catch (Exception e) {
161                     throw new EmbeddedJmxTransException("Exception converting settings " + outputWritersNode, e);
162                 }
163             }
164         } else {
165             logger.warn("Ignore invalid node {}", outputWritersNode);
166         }
167         return outputWriters;
168     }
169 
170     private OutputWriter instantiateOutputWriter(JsonNode outputWriterNode, String className) throws InstantiationException, IllegalAccessException, ClassNotFoundException, JsonProcessingException {
171         OutputWriterFactory<OutputWriter> factory = (OutputWriterFactory<OutputWriter>) Class.forName(className + "$Factory").newInstance();
172 
173         JsonNode settingsNode = outputWriterNode.path("settings");
174         Map<String, String> settings = Collections.emptyMap();
175         if (settingsNode.isMissingNode()) {
176         } else if (settingsNode.isObject()) {
177             ObjectMapper mapper = new ObjectMapper();
178             settings = mapper.treeToValue(settingsNode, Map.class);
179         } else {
180             logger.warn("Ignore invalid node {}", outputWriterNode);
181         }
182         return factory.create(settings);
183     }
184 
185     protected void parseQueryAttributeNode(@Nonnull Query query, @Nonnull JsonNode attributeNode) {
186         if (attributeNode.isMissingNode()) {
187         } else if (attributeNode.isValueNode()) {
188             query.addAttribute(attributeNode.asText());
189         } else if (attributeNode.isObject()) {
190             List<String> keys = null;
191 
192             JsonNode keysNode = attributeNode.path("keys");
193             if (keysNode.isMissingNode()) {
194             } else if (keysNode.isArray()) {
195                 if (keys == null) {
196                     keys = new ArrayList<>();
197                 }
198                 Iterator<JsonNode> itAttributeNode = keysNode.elements();
199                 while (itAttributeNode.hasNext()) {
200                     JsonNode keyNode = itAttributeNode.next();
201                     if (keyNode.isValueNode()) {
202                         keys.add(keyNode.asText());
203                     } else {
204                         logger.warn("Ignore invalid node {}", keyNode);
205                     }
206                 }
207             } else {
208                 logger.warn("Ignore invalid node {}", keysNode);
209             }
210 
211             JsonNode keyNode = attributeNode.path("key");
212             if (keyNode.isMissingNode()) {
213             } else if (keyNode.isValueNode()) {
214                 if (keys == null) {
215                     keys = new ArrayList<>();
216                 }
217                 keys.add(keyNode.asText());
218             } else {
219                 logger.warn("Ignore invalid node {}", keyNode);
220             }
221 
222             String name = attributeNode.path("name").asText();
223             JsonNode resultAliasNode = attributeNode.path("resultAlias");
224             String resultAlias = resultAliasNode.isMissingNode() ? null : resultAliasNode.asText();
225             JsonNode typeNode = attributeNode.path("type");
226             String type = typeNode.isMissingNode() ? null : typeNode.asText();
227             if (keys == null) {
228                 query.addAttribute(QueryAttribute.builder(name)
229                         .withType(type)
230                         .withResultAlias(resultAlias)
231                         .build());
232             } else {
233                 query.addAttribute(QueryAttribute.builder(name)
234                         .withType(type)
235                         .withResultAlias(resultAlias)
236                         .withKeys(keys)
237                         .build());
238             }
239         } else {
240             logger.warn("Ignore invalid node {}", attributeNode);
241         }
242     }
