149     /**
150      * Collect the values for this query and store them as {@link QueryResult} in the {@linkplain org.jmxtrans.embedded.Query#queryResults} queue
151      */
152     @Override
153     public void collectMetrics() {
154         long nanosBefore = System.nanoTime();
155         /*
156          * Optimisation tip: no need to skip 'mbeanServer.queryNames()' if the ObjectName is not a pattern
157          * (i.e. not '*' or '?' wildcard) because the mbeanserver internally performs the check.
158          * Seen on com.sun.jmx.interceptor.DefaultMBeanServerInterceptor
159          */
160         Set<ObjectName> matchingObjectNames = mbeanServer.queryNames(this.objectName, null);
161         logger.trace("Query {} returned {}", objectName, matchingObjectNames);
162 
163         for (ObjectName matchingObjectName : matchingObjectNames) {
164             long epochInMillis = System.currentTimeMillis();
165             try {
166                 AttributeList jmxAttributes = mbeanServer.getAttributes(matchingObjectName, this.attributeNames);
167                 logger.trace("Query {} returned {}", matchingObjectName, jmxAttributes);
168                 for (Attribute jmxAttribute : jmxAttributes.asList()) {
169                     QueryAttribute queryAttribute = this.attributesByName.get(jmxAttribute.getName());
170                     Object value = jmxAttribute.getValue();
171                     int count = queryAttribute.collectMetrics(matchingObjectName, value, epochInMillis, this.queryResults);
172                     collectedMetricsCount.addAndGet(count);
173                 }
174             } catch (Exception e) {
175                 logger.warn("Exception processing query {}", this, e);
176             }
177         }
178         collectionCount.incrementAndGet();
179         long nanosAfter = System.nanoTime();
180         collectionDurationInNanos.addAndGet(nanosAfter - nanosBefore);
181     }
182 
183     /**
184      * Export the collected metrics to the {@linkplain org.jmxtrans.embedded.output.OutputWriter}s associated with this {@linkplain org.jmxtrans.embedded.Query}
185      * (see {@link #getEffectiveOutputWriters()}).
186      * <p/>
187      * Metrics are batched according to {@link EmbeddedJmxTrans#getExportBatchSize()}
188      *
189      * @return the number of exported {@linkplain QueryResult}
190      */
191     @Override
192     public int exportCollectedMetrics() {
193         if(queryResults.isEmpty()) {
194             return 0;
195         }
196 
197         int totalExportedMetricsCount = 0;
198         long nanosBefore = System.nanoTime();
199 
200         List<OutputWriter> effectiveOutputWriters = getEffectiveOutputWriters();
201         int exportBatchSize = getEmbeddedJmxTrans().getExportBatchSize();
202         List<QueryResult> availableQueryResults = new ArrayList<QueryResult>(exportBatchSize);
203 
204         int size;
205         while ((size = queryResults.drainTo(availableQueryResults, exportBatchSize)) > 0) {
206             totalExportedMetricsCount += size;
207             exportedMetricsCount.addAndGet(size);
208             for (OutputWriter outputWriter : effectiveOutputWriters) {
209                 outputWriter.write(availableQueryResults);
210             }
211             availableQueryResults.clear();
212         }
213         exportDurationInNanos.addAndGet(System.nanoTime() - nanosBefore);
214         exportCount.incrementAndGet();
215         return totalExportedMetricsCount;
216     }

218     /**
219      * Start all the {@linkplain org.jmxtrans.embedded.output.OutputWriter}s attached to this {@linkplain org.jmxtrans.embedded.Query}
220      */
221     @PostConstruct
222     public void start() throws Exception {
223         queryMbeanObjectName = JmxUtils2.registerObject(this, "org.jmxtrans.embedded:Type=Query,id=" + id, getEmbeddedJmxTrans().getMbeanServer());
224 
225 
226         for (OutputWriter outputWriter : outputWriters) {
227             outputWriter.start();
228         }
229     }
230 
231     /**
232      * Stop all the {@linkplain org.jmxtrans.embedded.output.OutputWriter}s attached to this {@linkplain org.jmxtrans.embedded.Query}
233      */
234     @PreDestroy
235     public void stop() throws Exception {
236         JmxUtils2.unregisterObject(queryMbeanObjectName, mbeanServer);
237 
238         for (OutputWriter outputWriter : outputWriters) {
239             outputWriter.stop();
240         }
241     }

254     /**
255      * Add the given attribute to the list attributes of this query
256      * and maintains the reverse relation (see {@link org.jmxtrans.embedded.QueryAttribute#getQuery()}).
257      *
258      * @param attribute attribute to add
259      * @return this
260      */
261     @Nonnull
262     public Query addAttribute(@Nonnull QueryAttribute attribute) {
263         attribute.setQuery(this);
264         attributesByName.put(attribute.getName(), attribute);
265         attributeNames = attributesByName.keySet().toArray(new String[0]);
266 
267         return this;
268     }

307     /**
308      * Return the <code>outputWriters</code> to which the collected metrics of this {@linkplain org.jmxtrans.embedded.Query} are exported,
309      * the <code>outputWriters</code> declared at query level or a the parent level.
310      *
311      * @see #getOutputWriters()
312      * @see EmbeddedJmxTrans#getOutputWriters()
313      */
314     @Nonnull
315     public List<OutputWriter> getEffectiveOutputWriters() {
316         // Google Guava predicates would be nicer but we don't include guava to ease embeddability
317         List<OutputWriter> result = new ArrayList<OutputWriter>(embeddedJmxTrans.getOutputWriters().size() + outputWriters.size());
318         for (OutputWriter outputWriter : embeddedJmxTrans.getOutputWriters()) {
319             if (outputWriter.isEnabled()) {
320                 result.add(outputWriter);
321             }
322         }
323         for (OutputWriter outputWriter : outputWriters) {
324             if (outputWriter.isEnabled()) {
325                 result.add(outputWriter);
326             }
327         }
328         return result;
329     }

391     /**
392      * Returns the number of discarded elements in the {@link #queryResults} queue
393      * or <code>-1</code> if the queue is not a {@link org.jmxtrans.utils.concurrent.DiscardingBlockingQueue}.
394      */
395     @Override
396     public int getDiscardedResultsCount() {
397         if (queryResults instanceof DiscardingBlockingQueue) {
398             DiscardingBlockingQueue discardingBlockingQueue = (DiscardingBlockingQueue) queryResults;
399             return discardingBlockingQueue.getDiscardedElementCount();
400         } else {
401             return -1;
402         }
