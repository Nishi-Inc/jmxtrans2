23  package org.jmxtrans.example;
24  
25  import com.google.inject.Guice;
26  import com.google.inject.Injector;
27  
28  import org.jmxtrans.JmxTransformer;
29  import org.jmxtrans.cli.JmxTransConfiguration;
30  import org.jmxtrans.guice.JmxTransModule;
31  import org.jmxtrans.model.JmxProcess;
32  import org.jmxtrans.model.Query;
33  import org.jmxtrans.model.Server;
34  import org.jmxtrans.model.output.GraphiteWriter;
35  import org.jmxtrans.util.JsonPrinter;
36  
37  /**
38   * This example shows how to query ehcache for its statistics information.
39   * 
40   * @author jon
41   */
42  public class Ehcache {
43  
44  	private static final String GW_HOST = "192.168.192.133";
45  	private static final JsonPrinter printer = new JsonPrinter(System.out);
46  
47  	/** */
48  	public static void main(String[] args) throws Exception {
49  
50  		JmxProcess process = new JmxProcess(Server.builder()
51  				.setHost("w2")
52  				.setPort("1099")
53  				.setAlias("w2_ehcache_1099")
54  				.addQuery(Query.builder()
55  						.setObj("net.sf.ehcache:CacheManager=net.sf.ehcache.CacheManager@*,name=*,type=CacheStatistics")
56  						.addAttr("CacheHits")
57  						.addAttr("InMemoryHits")
58  						.addAttr("OnDiskHits")
59  						.addAttr("CacheMisses")
60  						.addAttr("ObjectCount")
61  						.addAttr("MemoryStoreObjectCount")
62  						.addAttr("DiskStoreObjectCount")
63  						.addOutputWriter(GraphiteWriter.builder()
64  								.addTypeName("name")
65  								.setDebugEnabled(true)
66  								.setHost(GW_HOST)
67  								.setPort(2003)
68  								.build())
69  						.build())
70  				.build());
71  
72  		printer.prettyPrint(process);
73  
74  		Injector injector = Guice.createInjector(new JmxTransModule(new JmxTransConfiguration()));
75  		JmxTransformer transformer = injector.getInstance(JmxTransformer.class);
76  
77  		transformer.executeStandalone(process);
78  	}
79  }
