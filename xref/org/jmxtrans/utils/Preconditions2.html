24  package org.jmxtrans.utils;
25  
26  import javax.annotation.Nullable;
27  
28  /**
29   * Inspired by <code>com.google.common.base.Preconditions</code>
30   *
31   * @author <a href="mailto:cleclerc@cloudbees.com">Cyrille Le Clerc</a>
32   */
33  public class Preconditions2 {
34  
35      public static <T> T checkNotNull(T t) {
36          if (t == null)
37              throw new NullPointerException();
38          return t;
39      }
40  
41      /**
42       * Check the nullity of the given <code>reference</code>.
43       *
44       * @param reference reference to check
45       * @param message   exception message, can be <code>null</code>
46       * @return the given <code>reference</code>
47       * @throws NullPointerException if the given <code>reference</code> is <code>null</code>
48       */
49      public static <T> T checkNotNull(T reference, @Nullable String message) throws NullPointerException {
50          if (reference == null) {
51              throw new NullPointerException(message == null ? "" : message);
52          }
53          return reference;
54      }
55  
56      public static String checkNotEmpty(String str) {
57          if (str == null)
58              throw new NullPointerException();
59          if (str.isEmpty())
60              throw new IllegalArgumentException("Can not be empty");
61          return str;
62      }
63  
64      public static void checkArgument(boolean expression) throws IllegalArgumentException {
65          if (!expression) {
66              throw new IllegalArgumentException();
67          }
68      }
69  
70      public static void checkArgument(boolean expression, @Nullable String msg) throws IllegalArgumentException {
71          if (!expression) {
72              throw new IllegalArgumentException(msg);
73          }
74      }
75  
76      /**
77       * @param expression
78       * @param msgFormat
79       * @param msgArgs
80       * @throws IllegalArgumentException
81       * @see String#format(String, Object...)
82       */
83      public static void checkArgument(boolean expression, @Nullable String msgFormat, Object... msgArgs) throws IllegalArgumentException {
84          if (!expression) {
85              if (msgFormat == null) {
86                  throw new IllegalArgumentException();
87              } else {
88                  throw new IllegalArgumentException(String.format(msgFormat, msgArgs));
89              }
90          }
91      }
92  }
