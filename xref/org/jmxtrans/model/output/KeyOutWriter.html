49  /**
50   * Writes out data in the same format as the GraphiteWriter, except to a file
51   * and tab delimited. Takes advantage of Log4J RollingFileAppender to
52   * automatically handle rolling the files after they reach a certain size.
53   * <p/>
54   * The default max size of the log files are 10MB (maxLogFileSize) The default
55   * number of rolled files to keep is 200 (maxLogBackupFiles)
56   *
57   * @author jon
58   */
59  public class KeyOutWriter extends BaseOutputWriter {
60  
61  	protected static final Log4jLoggerFactory log4jLoggerFactory = new Log4jLoggerFactory();
62  	protected static final String SETTING_MAX_LOG_FILE_SIZE = "maxLogFileSize";
63  	protected static final String SETTING_MAX_BACK_FILES = "maxLogBackupFiles";
64  	protected static final String SETTING_DELIMITER = "delimiter";
65  	protected static final String LOG_PATTERN = "%m%n";
66  	protected static final int LOG_IO_BUFFER_SIZE_BYTES = 1024;
67  	protected static final Map<String, Logger> loggers = new ConcurrentHashMap<String, Logger>();
68  
69  	protected static final int MAX_LOG_BACKUP_FILES = 200;
70  	protected static final String MAX_LOG_FILE_SIZE = "10MB";
71  	protected static final String DEFAULT_DELIMITER = "\t";
72  	protected Logger logger;
73  
74  
75  	private final String outputFile;
76  	private final String maxLogFileSize;
77  	private final int maxLogBackupFiles;
78  	private final String delimiter;
79  
80  	@JsonCreator
81  	public KeyOutWriter(
82  			@JsonProperty("typeNames") ImmutableList<String> typeNames,
83  			@JsonProperty("booleanAsNumber") boolean booleanAsNumber,
84  			@JsonProperty("debug") Boolean debugEnabled,
85  			@JsonProperty("outputFile") String outputFile,
86  			@JsonProperty("maxLogFileSize") String maxLogFileSize,
87  			@JsonProperty("maxLogBackupFiles") int maxLogBackupFiles,
88  			@JsonProperty("delimiter") String delimiter,
89  			@JsonProperty("settings") Map<String, Object> settings) {
90  		super(typeNames, booleanAsNumber, debugEnabled, settings);
91  		this.outputFile = MoreObjects.firstNonNull(
92  				outputFile,
93  				(String) getSettings().get("outputFile"));
94  		this.maxLogFileSize = firstNonNull(
95  				maxLogFileSize,
96  				(String) getSettings().get(SETTING_MAX_LOG_FILE_SIZE),
97  				MAX_LOG_FILE_SIZE);
98  		this.maxLogBackupFiles = firstNonNull(
99  				maxLogBackupFiles,
100 				(Integer) getSettings().get(SETTING_MAX_BACK_FILES),
101 				MAX_LOG_BACKUP_FILES);
102 		this.delimiter = firstNonNull(
103 				delimiter,
104 				(String) getSettings().get(SETTING_DELIMITER),
105 				DEFAULT_DELIMITER
106 		);
107 	}
108 
109 	/**
110 	 * Creates the logging
111 	 */
112 	@Override
113 	public void validateSetup(Server server, Query query) throws ValidationException {
114 		// Check if we've already created a logger for this file. If so, use it.
115 		if (loggers.containsKey(outputFile)) {
116 			logger = loggers.get(outputFile);
117 			return;
118 		}
119 		// need to create a logger
120 		try {
121 			logger = initLogger(outputFile);
122 			loggers.put(outputFile, logger);
123 		} catch (IOException e) {
124 			throw new ValidationException("Failed to setup log4j", query);
125 		}
126 	}
127 
128 	/**
129 	 * The meat of the output. Very similar to GraphiteWriter.
130 	 */
131 	@Override
132 	public void internalWrite(Server server, Query query, ImmutableList<Result> results) throws Exception {
133 		List<String> typeNames = getTypeNames();
134 
135 		for (Result result : results) {
136 			Map<String, Object> resultValues = result.getValues();
137 			if (resultValues != null) {
138 				for (Entry<String, Object> values : resultValues.entrySet()) {
139 					if (NumberUtils.isNumeric(values.getValue())) {
140 
141 						logger.info(KeyUtils.getKeyString(server, query, result, values, typeNames, null) + delimiter
142 								+ values.getValue().toString() + delimiter + result.getEpoch());
143 					}
144 				}
145 			}
146 		}
147 	}
148 
149 	/**
150 	 * Initializes the logger. This is called when we need to create a new
151 	 * logger for the given file name.
152 	 *
153 	 * @param fileStr
154 	 * @return a new Logger instance for the given fileStr
155 	 * @throws java.io.IOException
156 	 */
157 	protected Logger initLogger(String fileStr) throws IOException {
158 		Appender appender = buildLog4jAppender(fileStr, getMaxLogFileSize(), getMaxLogBackupFiles());
159 		LoggerFactory loggerFactory = buildLog4jLoggerFactory(appender);
160 
161 		String loggerKey = buildLoggerName();
162 
163 		// Create the logger and add to the map of loggers using our factory
164 		LogManager.getLogger(loggerKey, loggerFactory);
165 		return log4jLoggerFactory.getLogger(loggerKey);
166 	}
167 
168 	protected String buildLoggerName() {
169 		return "KeyOutWriter" + this.hashCode();
170 	}
171 
172 	protected Appender buildLog4jAppender(
173 			String fileStr, String maxLogFileSize,
174 			Integer maxLogBackupFiles) throws IOException {
175 
176 		PatternLayout pl = new PatternLayout(LOG_PATTERN);
177 		final RollingFileAppender appender = new RollingFileAppender(pl, fileStr, true);
178 		appender.setImmediateFlush(true);
179 		appender.setBufferedIO(false);
180 		appender.setBufferSize(LOG_IO_BUFFER_SIZE_BYTES);
181 		appender.setMaxFileSize(maxLogFileSize);
182 		appender.setMaxBackupIndex(maxLogBackupFiles);
183 
184 		return appender;
185 	}
186 
187 	protected LoggerFactory buildLog4jLoggerFactory(final Appender appender) {
188 		return new LoggerFactory() {
189 			@Override
190 			public org.apache.log4j.Logger makeNewLoggerInstance(String name) {
191 				org.apache.log4j.Logger logger = org.apache.log4j.Logger.getLogger(name);
192 				logger.addAppender(appender);
193 				logger.setLevel(org.apache.log4j.Level.INFO);
194 				logger.setAdditivity(false);
195 				return logger;
196 			}
197 		};
198 	}
199 
200 	public String getMaxLogFileSize() {
201 		return maxLogFileSize;
202 	}
203 
204 	public Integer getMaxLogBackupFiles() {
205 		return maxLogBackupFiles;
206 	}
207 
208 	public String getDelimiter() {
209 		return delimiter;
210 	}
211 
212 	public String getOutputFile() {
213 		return outputFile;
214 	}
