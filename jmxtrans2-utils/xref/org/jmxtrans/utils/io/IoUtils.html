36  public final class IoUtils {
37      private static final Logger LOGGER = Logger.getLogger(IoUtils.class.getName());
38  
39      private static final int COPY_BUFFER_SIZE = 512;
40  
41      /**
42       * Simple implementation without chunking if the source file is big.
43       *
44       * @param source
45       * @param destination
46       * @throws java.io.IOException
47       */
48      public static void doCopySmallFile(@Nonnull File source, @Nonnull File destination, boolean append) throws IOException {
49          if (destination.exists() && destination.isDirectory()) {
50              throw new IOException("Can not copy file, destination is a directory: " + destination.getAbsolutePath());
51          } else if (!destination.exists()) {
52              boolean renamed = source.renameTo(destination);
53              if (renamed) return;
54          }
55  
56          FileOutputStream fos = null;
57          try {
58              fos = new FileOutputStream(destination, append);
59              if (append) {
60                  fos.write("\n".getBytes("UTF-8"));
61              }
62              fos.write(Files.readAllBytes(Paths.get(source.getAbsolutePath())));
63          } finally {
64              if (fos != null) {
65                  fos.close();
66              }
67          }
68      }
69  
70      /**
71       * Simple implementation without chunking if the source file is big.
72       *
73       * @param source
74       * @param destination
75       * @throws java.io.IOException
76       */
77      private static void doCopySmallFile(@Nonnull File source, @Nonnull File destination) throws IOException {
78          if (destination.exists() && destination.isDirectory()) {
79              throw new IOException("Can not copy file, destination is a directory: " + destination.getAbsolutePath());
80          }
81  
82          FileInputStream fis = null;
83          FileOutputStream fos = null;
84          FileChannel input = null;
85          FileChannel output = null;
86          try {
87              fis = new FileInputStream(source);
88              fos = new FileOutputStream(destination, false);
89              input = fis.getChannel();
90              output = fos.getChannel();
91              output.transferFrom(input, 0, input.size());
92          } finally {
93              closeQuietly(output);
94              closeQuietly(input);
95              closeQuietly(fis);
96              closeQuietly(fos);
97          }
98  
99          if (destination.length() != source.length()) {
100             throw new IOException("Failed to copy content from '" +
101                     source + "' (" + source.length() + "bytes) to '" + destination + "' (" + destination.length() + ")");
102         }
103     }

105     public static void appendToFile(File source, File destination, long maxFileSize, int maxBackupIndex) throws IOException {
106         boolean destinationExists = validateDestinationFile(source, destination, maxFileSize, maxBackupIndex);
107         if (destinationExists) {
108             doCopySmallFile(source, destination, true);
109         } else {
110             boolean renamed = source.renameTo(destination);
111             if (!renamed) {
112                 doCopySmallFile(source, destination, false);
113             }
114         }
115     }
116 
117     private static boolean validateDestinationFile(File source, File destination, long maxFileSize, int maxBackupIndex) throws IOException {
118         if (!destination.exists() || destination.isDirectory()) return false;
119         long totalLengthAfterAppending = destination.length() + source.length();
120         if (totalLengthAfterAppending > maxFileSize) {
121             rollFiles(destination, maxBackupIndex);
122             return false; // File no longer exists because it was move to filename.1
123         }
124 
125         return true;
126     }
127 
128     private static void rollFiles(File destination, int maxBackupIndex) throws IOException {
129 
130         // if maxBackup index == 10 then we will have file
131         // outputFile, outpuFile.1 outputFile.2 ... outputFile.10
132         // we only care if 9 and lower exists to move them up a number
133         for (int i = maxBackupIndex - 1; i >= 0; i--) {
134             String path = destination.getAbsolutePath();
135             path=(i==0)?path:path + "." + i;
136             File f = new File(path);
137             if (!f.exists()) continue;
138 
139             File fNext = new File(destination + "." + (i + 1));
140             doCopySmallFile(f, fNext, false);
141         }
142 
143         if (!destination.delete()) {
144             LOGGER.log(Level.WARNING, "Could not delete file [" + destination.getAbsolutePath() + "].");
145         }
146     }
147 
148     public static void replaceFile(File source, File destination) throws IOException {
149         if (destination.exists()) {
150             // try to delete destination, it might fail, but doCopySmallFile() can deal with it
151             if (!destination.delete()) {
152                 LOGGER.info("Could not delete " + destination.getAbsolutePath() + ", replacing only the content");
153             }
154         }
155         doCopySmallFile(source, destination);
156     }

158     public static void copy(InputStream in, OutputStream out) throws IOException {
159         byte[] buffer = new byte[COPY_BUFFER_SIZE];
160         int len;
161         while ((len = in.read(buffer)) != -1) {
162             out.write(buffer, 0, len);
163         }
164     }

166     public static OutputStream nullOutputStream() {
167         return new NullOutputStream();
168     }

170     public static void closeQuietly(Closeable closeable) {
171         if (closeable == null)
172             return;
173         try {
174             closeable.close();
175         } catch (IOException e) {
176             // Not being able to close something is still a problem : potential data loss, not releasing
177             // resources, ... So we should still log it.
178             LOGGER.log(Level.WARNING, "Could not close object");
179         }
180     }
181 
182     public static void closeQuietly(Writer writer) {
183         if (writer == null)
184             return;
185         try {
186             writer.close();
187         } catch (IOException e) {
188             // Not being able to close something is still a problem : potential data loss, not releasing
189             // resources, ... So we should still log it.
190             LOGGER.log(Level.WARNING, "Could not close writer");
191         }
192     }
193 
194     /**
195      * Needed for old JVMs where {@link java.io.InputStream} does not implement {@link java.io.Closeable}.
196      */
197     public static void closeQuietly(InputStream inputStream) {
198         if (inputStream == null)
199             return;
200         try {
201             inputStream.close();
202         } catch (IOException e) {
203             // Not being able to close something is still a problem : potential data loss, not releasing
204             // resources, ... So we should still log it.
205             LOGGER.log(Level.WARNING, "Could not close input stream");
206         }
207     }
