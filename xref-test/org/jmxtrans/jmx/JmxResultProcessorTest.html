52  /**
53   * <b>Warning:</b> This test class relies on MBeans exposed by the JVM. As far
54   * as I understand the documentation, it only relies on beans and attributes
55   * that should be present on all implementations. Still there is some chance
56   * that this test will fail on some JVMs or that it depends too much on
57   * specific platform properties.
58   */
59  public class JmxResultProcessorTest {
60  
61  	private Query query;
62  
63  	@Before
64  	public void initConfiguration() {
65  		query = Query.builder()
66  				.setResultAlias("resultAlias")
67  				.build();
68  	}
69  
70  	@Test
71  	public void canCreateBasicResultData() throws MalformedObjectNameException, InstanceNotFoundException {
72  		Attribute integerAttribute = new Attribute("StartTime", 51L);
73  		ObjectInstance runtime = getRuntime();
74  		List<Result> results = new JmxResultProcessor(
75  				query,
76  				runtime,
77  				ImmutableList.of(integerAttribute),
78  				runtime.getClassName()).getResults();
79  
80  		assertThat(results).hasSize(1);
81  		Result integerResult = results.get(0);
82  
83  		assertThat(integerResult.getAttributeName()).isEqualTo("StartTime");
84  		assertThat(integerResult.getClassName()).isEqualTo("sun.management.RuntimeImpl");
85  		assertThat(integerResult.getClassNameAlias()).isEqualTo("resultAlias");
86  		assertThat(integerResult.getTypeName()).isEqualTo("type=Runtime");
87  		assertThat(integerResult.getValues()).hasSize(1);
88  	}

90  	@Test
91  	public void canReadSingleIntegerValue() throws MalformedObjectNameException, InstanceNotFoundException {
92  		Attribute integerAttribute = new Attribute("CollectionCount", 51L);
93  		ObjectInstance runtime = getRuntime();
94  		List<Result> results = new JmxResultProcessor(
95  				query,
96  				runtime,
97  				ImmutableList.of(integerAttribute),
98  				runtime.getClassName()).getResults();
99  
100 		assertThat(results).hasSize(1);
101 		Result integerResult = results.get(0);
102 
103 		assertThat(integerResult.getAttributeName()).isEqualTo("CollectionCount");
104 		assertThat(integerResult.getValues()).hasSize(1);
105 
106 		Object objectValue = integerResult.getValues().get("CollectionCount");
107 		assertThat(objectValue).isInstanceOf(Long.class);
108 
109 		Long integerValue = (Long) objectValue;
110 		assertThat(integerValue).isEqualTo(51L);
111 	}

136 	@Test
137 	public void canReadTabularData() throws MalformedObjectNameException, AttributeNotFoundException, MBeanException,
138 			ReflectionException, InstanceNotFoundException {
139 		ObjectInstance runtime = getRuntime();
140 		AttributeList attr = ManagementFactory.getPlatformMBeanServer().getAttributes(
141 				runtime.getObjectName(), new String[]{"SystemProperties"});
142 		List<Result> results = new JmxResultProcessor(
143 				query,
144 				runtime,
145 				attr.asList(),
146 				runtime.getClassName()).getResults();
147 
148 		assertThat(results.size()).isGreaterThan(2);
149 
150 		Optional<Result> result = from(results).firstMatch(new ByAttributeName("SystemProperties.java.version"));
151 
152 		assertThat(result.isPresent()).isTrue();
153 
154 		assertThat(result.get().getAttributeName()).isEqualTo("SystemProperties.java.version");
155 		Map<String,Object> values = result.get().getValues();
156 		assertThat(values).hasSize(2);
157 
158 		Object objectValue = result.get().getValues().get("key");
159 		assertThat(objectValue).isInstanceOf(String.class);
160 
161 		String key = (String) objectValue;
162 		assertThat(key).isEqualTo("java.version");
163 	}

201 	private static class ByAttributeName implements Predicate<Result> {
202 		private final String attributeName;
203 
204 		public ByAttributeName(String attributeName) {
205 			this.attributeName = attributeName;
206 		}
207 
208 		@Override
209 		public boolean apply(@Nullable Result result) {
210 			if (result == null) {
211 				return false;
212 			}
213 			return attributeName.equals(result.getAttributeName());
214 		}
