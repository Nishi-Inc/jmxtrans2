23  package org.jmxtrans.embedded.query;
24  
25  import org.jmxtrans.embedded.MockMemoryPool;
26  import org.jmxtrans.results.QueryResult;
27  import org.jmxtrans.utils.concurrent.DiscardingBlockingQueue;
28  import org.junit.AfterClass;
29  import org.junit.BeforeClass;
30  import org.junit.Test;
31  
32  import javax.management.MBeanServer;
33  import javax.management.ObjectName;
34  import java.lang.management.ManagementFactory;
35  import java.util.Arrays;
36  import java.util.concurrent.BlockingQueue;
37  
38  import static org.assertj.core.api.Assertions.assertThat;
39  
40  /**
41   * @author <a href="mailto:cleclerc@xebia.fr">Cyrille Le Clerc</a>
42   */
43  public class QueryTest {
44  
45      static MBeanServer mbeanServer = ManagementFactory.getPlatformMBeanServer();
46      static ObjectName mockEdenSpacePool;
47      static ObjectName mockPermGenPool;
48  
49  
50      @BeforeClass
51      public static void beforeClass() throws Exception {
52          mockEdenSpacePool = new ObjectName("test:type=MemoryPool,name=PS Eden Space");
53          mbeanServer.registerMBean(new MockMemoryPool("PS Eden Space", 87359488L), mockEdenSpacePool);
54          mockPermGenPool = new ObjectName("test:type=MemoryPool,name=PS Perm Gen");
55          mbeanServer.registerMBean(new MockMemoryPool("PS Perm Gen", 87752704L), mockPermGenPool);
56      }
57  
58      @AfterClass
59      public static void afterClass() throws Exception {
60          mbeanServer.unregisterMBean(mockEdenSpacePool);
61          mbeanServer.unregisterMBean(mockPermGenPool);
62      }
63  
64      @Test
65      public void basic_jmx_attribute_return_simple_result() throws Exception {
66          Query query = new Query("test:type=MemoryPool,name=PS Eden Space").addAttribute("CollectionUsageThreshold");
67          BlockingQueue<QueryResult> results = new DiscardingBlockingQueue<QueryResult>(10);
68          query.collectMetrics(mbeanServer, results);
69          assertThat(results).hasSize(1);
70  
71          QueryResult result = results.poll();
72          assertThat(result.getValue()).isInstanceOf(Number.class);
73      }
74  
75      @Test
76      public void test_composite_jmx_attribute() throws Exception {
77          Query query = new Query("test:type=MemoryPool,name=PS Perm Gen");
78          query.addAttribute(QueryAttribute.builder("Usage")
79                  .withKeys(Arrays.asList("committed", "init", "max", "used"))
80                  .build());
81          BlockingQueue<QueryResult> results = new DiscardingBlockingQueue<QueryResult>(10);
82          query.collectMetrics(mbeanServer, results);
83          assertThat(results).hasSize(4);
84  
85          QueryResult result1 = results.poll();
86          assertThat(result1.getValue()).isInstanceOf(Number.class);
87  
88          QueryResult result2 = results.poll();
89          assertThat(result2.getValue()).isInstanceOf(Number.class);
90      }
91  }
