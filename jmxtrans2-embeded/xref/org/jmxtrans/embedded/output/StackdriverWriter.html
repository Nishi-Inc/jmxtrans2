52  /**
53   * <a href="https://www.stackdriver.com//">Stackdriver</a> implementation of the
54   * {@linkplain OutputWriter}.
55   * <p/>
56   * This implementation uses <a href="https://custom-gateway.stackdriver.com/v1/custom"> POST {@code /v1/metrics}</a>
57   * HTTP API.
58   * <p/>
59   * Settings:
60   * <ul>
61   * <li>"{@code url}": Stackdriver server URL. Optional, default value: {@value #DEFAULT_STACKDRIVER_API_URL}.</li>
62   * <li>"{@code token}": Stackdriver API token. Mandatory</li>
63   * <li>"{@code source}": Instance of the machine ID that the JMX data is being collected from. Optional.
64   * <li>"{@code detectInstance}": Set to "AWS" if you want to detect the local AWS instance ID on startup.  Optional. 
65   * <li>"{@code stackdriverApiTimeoutInMillis}": read timeout of the calls to Stackdriver HTTP API. Optional, default
66   * value: {@value #DEFAULT_STACKDRIVER_API_TIMEOUT_IN_MILLIS}.</li>
67   * <li>"{@code enabled}": flag to enable/disable the writer. Optional, default value: <code>true</code>.</li>
68   * </ul>
69   * 
70   * @author <a href="mailto:eric@stackdriver.com">Eric Kilby</a>
71   */
72  public class StackdriverWriter extends AbstractOutputWriter implements OutputWriter {
73  
74  	public static final int STACKDRIVER_PROTOCOL_VERSION = 1;
75  	public static final String SETTING_SOURCE_INSTANCE = "source";
76  	public static final String SETTING_DETECT_INSTANCE = "detectInstance";
77  	public static final String DEFAULT_STACKDRIVER_API_URL = "https://custom-gateway.stackdriver.com/v1/custom";
78  	public static final String SETTING_STACKDRIVER_API_TIMEOUT_IN_MILLIS = "stackdriverApiTimeoutInMillis";
79  	public static final int DEFAULT_STACKDRIVER_API_TIMEOUT_IN_MILLIS = 1000;
80  	private final Logger logger = LoggerFactory.getLogger(getClass());
81  	private final AtomicInteger exceptionCounter = new AtomicInteger();
82  	private JsonFactory jsonFactory = new JsonFactory();
83  	/**
84  	 * Stackdriver HTTP API URL
85  	 */
86  	private URL url;
87  	private int stackdriverApiTimeoutInMillis = DEFAULT_STACKDRIVER_API_TIMEOUT_IN_MILLIS;
88  
89  	/**
90  	 * Stackdriver API key
91  	 */
92  	private String apiKey;
93  
94  	/**
95  	 * Optional proxy for the http API calls
96  	 */
97  	@Nullable
98  	private Proxy proxy;
99  
100 	/**
101 	 * AWS instance ID, populated on startup
102 	 */
103     @Nullable
104 	private String instanceId;

106 	/**
107 	 * Initial setup for the writer class. Loads in settings and initializes one-time setup variables like instanceId.
108 	 */
109 	@Override
110 	public void start() {
111 		try {
112 			url = new URL(getStringSetting(SETTING_URL, DEFAULT_STACKDRIVER_API_URL));
113 		} catch (MalformedURLException e) {
114 			throw new EmbeddedJmxTransException(e);
115 		}
116 
117 		apiKey = getStringSetting(SETTING_TOKEN);
118 
119 		if (getStringSetting(SETTING_PROXY_HOST, null) != null && !getStringSetting(SETTING_PROXY_HOST).isEmpty()) {
120 			proxy = new Proxy(Proxy.Type.HTTP, new InetSocketAddress(getStringSetting(SETTING_PROXY_HOST), getIntSetting(SETTING_PROXY_PORT)));
121 		}
122 		
123 		logger.info("Starting Stackdriver writer connected to '{}', proxy {} ...", url, proxy);
124 		
125 
126 		stackdriverApiTimeoutInMillis = getIntSetting(SETTING_STACKDRIVER_API_TIMEOUT_IN_MILLIS, DEFAULT_STACKDRIVER_API_TIMEOUT_IN_MILLIS);
127 
128 		// try to get and instance ID
129 		if (getStringSetting(SETTING_SOURCE_INSTANCE, null) != null && !getStringSetting(SETTING_SOURCE_INSTANCE).isEmpty()) {
130 			// if one is set directly use that
131 			instanceId = getStringSetting(SETTING_SOURCE_INSTANCE);
132 			logger.info("Using instance ID {} from setting {}", instanceId, SETTING_SOURCE_INSTANCE);
133 		} else if (getStringSetting(SETTING_DETECT_INSTANCE, null) != null && "AWS".equalsIgnoreCase(getStringSetting(SETTING_DETECT_INSTANCE))) {
134 			// if setting is to detect, look on the local machine URL
135 			logger.info("Detect instance set to AWS, trying to determine AWS instance ID");
136 			instanceId = getLocalAwsInstanceId();
137 			if (instanceId != null) {
138 				logger.info("Detected instance ID as {}", instanceId);
139 			} else {
140 				logger.info("Unable to detect AWS instance ID for this machine, sending metrics without an instance ID");
141 			}
142 		} else {
143 			// no instance ID, the metrics will be sent as "bare" custom metrics and not associated with an instance
144 			instanceId = null;
145 			logger.info("No source instance ID passed, and not set to detect, sending metrics without and instance ID");
146 		}
147 		
148 	}

150 	/**
151 	 * Send given metrics to the Stackdriver server using HTTP
152 	 * 
153 	 * @param results
154 	 *            Iterable collection of data points
155 	 */
156 	@Override
157 	public void write(Iterable<QueryResult> results) {
158 		logger.debug("Export to '{}', proxy {} metrics {}", url, proxy, results);
159 
160 		HttpURLConnection urlConnection = null;
161 
162 		try {
163 			if (proxy == null) {
164 				urlConnection = (HttpURLConnection) url.openConnection();
165 			} else {
166 				urlConnection = (HttpURLConnection) url.openConnection(proxy);
167 			}
168 			urlConnection.setRequestMethod("POST");
169 			urlConnection.setDoInput(true);
170 			urlConnection.setDoOutput(true);
171 			urlConnection.setReadTimeout(stackdriverApiTimeoutInMillis);
172 			urlConnection.setRequestProperty("content-type", "application/json; charset=utf-8");
173 			urlConnection.setRequestProperty("x-stackdriver-apikey", apiKey);
174 
175 			serialize(results, urlConnection.getOutputStream());
176 			int responseCode = urlConnection.getResponseCode();
177 			if (responseCode != 200 && responseCode != 201) {
178 				exceptionCounter.incrementAndGet();
179 				logger.warn("Failure {}:'{}' to send result to Stackdriver server '{}' with proxy {}", responseCode,
180 						urlConnection.getResponseMessage(), url, proxy);
181 			}
182 			if (logger.isTraceEnabled()) {
183 				IoUtils2.copy(urlConnection.getInputStream(), System.out);
184 			}
185 		} catch (Exception e) {
186 			exceptionCounter.incrementAndGet();
187 			logger.warn("Failure to send result to Stackdriver server '{}' with proxy {}", url, proxy, e);
188 		} finally {
189 			if (urlConnection != null) {
190 				try {
191 					InputStream in = urlConnection.getInputStream();
192 					IoUtils2.copy(in, IoUtils2.nullOutputStream());
193 					IoUtils2.closeQuietly(in);
194 					InputStream err = urlConnection.getErrorStream();
195 					if (err != null) {
196 						IoUtils2.copy(err, IoUtils2.nullOutputStream());
197 						IoUtils2.closeQuietly(err);
198 					}
199 					urlConnection.disconnect();
200 				} catch (IOException e) {
201 					logger.warn("Error flushing http connection for one result, continuing");
202 					logger.debug("Stack trace for the http connection, usually a network timeout", e);
203 				}
204 			}
205 
206 		}
207 	}

209 	/**
210 	 * Use the EC2 Metadata URL to determine the instance ID that this code is running on. Useful if you don't want to
211 	 * configure the instance ID manually. Pass detectInstance = "AWS" to have this run in your configuration.
212 	 * 
213 	 * @return String containing an AWS instance id, or null if none is found
214 	 */
215     @Nullable
216 	private String getLocalAwsInstanceId() {
217 		String detectedInstanceId = null;
218 		try {
219 			final URL metadataUrl = new URL("http://169.254.169.254/latest/meta-data/instance-id");
220             URLConnection metadataConnection = metadataUrl.openConnection();
221             BufferedReader in = new BufferedReader(new InputStreamReader(metadataConnection.getInputStream(), "UTF-8"));
222             String inputLine;
223             while ((inputLine = in.readLine()) != null) {
224 				detectedInstanceId = inputLine;
225 			}
226 			in.close();
227 		} catch (Exception e) {
228 			logger.warn("unable to determine AWS instance ID", e);
229 		}
230 		return detectedInstanceId;
231 	}

233 	/**
234 	 * Put the values into the JSON format expected by the Stackdriver custom metrics gateway
235 	 *
236      * @param results
237      *            Iterable collection of data points (gauges and counters)
238 	 * @param out
239 	 *            OutputStream to write JSON to
240 	 * @throws java.io.IOException
241 	 */
242 	public void serialize(@Nonnull Iterable<QueryResult> results, @Nonnull OutputStream out) throws IOException {
243 		JsonGenerator g = jsonFactory.createGenerator(out, JsonEncoding.UTF8);
244 		g.writeStartObject();
245 		g.writeNumberField("timestamp", System.currentTimeMillis() / 1000);
246 		g.writeNumberField("proto_version", STACKDRIVER_PROTOCOL_VERSION);
247 		g.writeArrayFieldStart("data");
248 
249 		for (QueryResult metric : results) {
250 			g.writeStartObject();
251 			g.writeStringField("name", metric.getName());
252 			if (instanceId != null && !instanceId.isEmpty()) {
253 				g.writeStringField("instance", instanceId);
254 			}
255 			g.writeNumberField("collected_at", metric.getEpoch(TimeUnit.SECONDS));
256 			if (metric.getValue() instanceof Integer) {
257 				g.writeNumberField("value", (Integer) metric.getValue());
258 			} else if (metric.getValue() instanceof Long) {
259 				g.writeNumberField("value", (Long) metric.getValue());
260 			} else if (metric.getValue() instanceof Float) {
261 				g.writeNumberField("value", (Float) metric.getValue());
262 			} else if (metric.getValue() instanceof Double) {
263 				g.writeNumberField("value", (Double) metric.getValue());
264 			}
265 			g.writeEndObject();
266 		}
267 
268 		g.writeEndArray();
269 		g.writeEndObject();
270 		g.flush();
271 		g.close();
272 	}
273 
274 	public int getExceptionCounter() {
275 		return exceptionCounter.get();
276 	}
