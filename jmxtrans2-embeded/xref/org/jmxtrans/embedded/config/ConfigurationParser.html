103     protected void mergeEmbeddedJmxTransConfiguration(@Nonnull String configurationUrl, @Nonnull EmbeddedJmxTrans embeddedJmxTrans) throws EmbeddedJmxTransException {
104         try {
105             if (configurationUrl.startsWith("classpath:")) {
106                 logger.debug("mergeEmbeddedJmxTransConfiguration({})", configurationUrl);
107                 String path = configurationUrl.substring("classpath:".length());
108                 InputStream in = Thread.currentThread().getContextClassLoader().getResourceAsStream(path);
109                 Preconditions.checkNotNull(in, "No file found for '" + configurationUrl + "'");
110                 mergeEmbeddedJmxTransConfiguration(in, embeddedJmxTrans);
111             } else {
112                 mergeEmbeddedJmxTransConfiguration(new URL(configurationUrl), embeddedJmxTrans);
113             }
114         } catch (JsonProcessingException e) {
115             throw new EmbeddedJmxTransException("Exception loading configuration'" + configurationUrl + "': " + e.getMessage(), e);
116         } catch (Exception e) {
117             throw new EmbeddedJmxTransException("Exception loading configuration'" + configurationUrl + "'", e);
118         }
119     }

145     protected void mergeEmbeddedJmxTransConfiguration(@Nonnull URL configurationUrl, EmbeddedJmxTrans embeddedJmxTrans) throws IOException {
146         logger.debug("mergeEmbeddedJmxTransConfiguration({})", configurationUrl);
147         JsonNode configurationRootNode = mapper.readValue(configurationUrl, JsonNode.class);
148         mergeEmbeddedJmxTransConfiguration(configurationRootNode, embeddedJmxTrans);
149     }

151     private void mergeEmbeddedJmxTransConfiguration(@Nonnull JsonNode configurationRootNode, @Nonnull EmbeddedJmxTrans embeddedJmxTrans) {
152         for (JsonNode queryNode : configurationRootNode.path("queries")) {
153 
154             String objectName = queryNode.path("objectName").asText();
155             Query query = new Query(objectName);
156             embeddedJmxTrans.addQuery(query);
157             JsonNode resultAliasNode = queryNode.path("resultAlias");
158             if (resultAliasNode.isMissingNode()) {
159             } else if (resultAliasNode.isValueNode()) {
160                 query.setResultAlias(resultAliasNode.asText());
161             } else {
162                 logger.warn("Ignore invalid node {}", resultAliasNode);
163             }
164 
165             JsonNode attributesNode = queryNode.path("attributes");
166             if (attributesNode.isMissingNode()) {
167             } else if (attributesNode.isArray()) {
168                 Iterator<JsonNode> itAttributeNode = attributesNode.elements();
169                 while (itAttributeNode.hasNext()) {
170                     JsonNode attributeNode = itAttributeNode.next();
171                     parseQueryAttributeNode(query, attributeNode);
172                 }
173             } else {
174                 logger.warn("Ignore invalid node {}", resultAliasNode);
175             }
176 
177             JsonNode attributeNode = queryNode.path("attribute");
178             parseQueryAttributeNode(query, attributeNode);
179             List<OutputWriter> outputWriters = parseOutputWritersNode(queryNode);
180             query.getOutputWriters().addAll(outputWriters);
181             logger.trace("Add {}", query);
182         }
183 
184         List<OutputWriter> outputWriters = parseOutputWritersNode(configurationRootNode);
185         embeddedJmxTrans.getOutputWriters().addAll(outputWriters);
186         logger.trace("Add global output writers: {}", outputWriters);
187 
188         JsonNode queryIntervalInSecondsNode = configurationRootNode.path("queryIntervalInSeconds");
189         if (!queryIntervalInSecondsNode.isMissingNode()) {
190             embeddedJmxTrans.setQueryIntervalInSeconds(queryIntervalInSecondsNode.asInt());
191         }
192 
193         JsonNode exportBatchSizeNode = configurationRootNode.path("exportBatchSize");
194         if (!exportBatchSizeNode.isMissingNode()) {
195             embeddedJmxTrans.setExportBatchSize(exportBatchSizeNode.asInt());
196         }
197 
198         JsonNode numQueryThreadsNode = configurationRootNode.path("numQueryThreads");
199         if (!numQueryThreadsNode.isMissingNode()) {
200             embeddedJmxTrans.setNumQueryThreads(numQueryThreadsNode.asInt());
201         }
202 
203         JsonNode exportIntervalInSecondsNode = configurationRootNode.path("exportIntervalInSeconds");
204         if (!exportIntervalInSecondsNode.isMissingNode()) {
205             embeddedJmxTrans.setExportIntervalInSeconds(exportIntervalInSecondsNode.asInt());
206         }
207 
208         JsonNode numExportThreadsNode = configurationRootNode.path("numExportThreads");
209         if (!numExportThreadsNode.isMissingNode()) {
210             embeddedJmxTrans.setNumExportThreads(numExportThreadsNode.asInt());
211         }
212 
213         logger.info("Loaded {}", embeddedJmxTrans);
214     }

216     private List<OutputWriter> parseOutputWritersNode(@Nonnull JsonNode outputWritersParentNode) {
217         JsonNode outputWritersNode = outputWritersParentNode.path("outputWriters");
218         List<OutputWriter> outputWriters = new ArrayList<OutputWriter>();
219         if (outputWritersNode.isMissingNode()) {
220         } else if (outputWritersNode.isArray()) {
221             for (JsonNode outputWriterNode : outputWritersNode) {
222                 try {
223                     String className = outputWriterNode.path("@class").asText();
224                     OutputWriter outputWriter = (OutputWriter) Class.forName(className).newInstance();
225                     JsonNode deprecatedEnabledNode = outputWriterNode.path("enabled");
226                     if (!deprecatedEnabledNode.isMissingNode()) {
227                         logger.warn("OutputWriter {}, deprecated usage of attribute 'enabled', settings{ \"enabled\":... } should be used instead");
228                         outputWriter.setEnabled(deprecatedEnabledNode.asBoolean());
229                     }
230                     JsonNode settingsNode = outputWriterNode.path("settings");
231                     if (settingsNode.isMissingNode()) {
232                     } else if (settingsNode.isObject()) {
233                         ObjectMapper mapper = new ObjectMapper();
234                         @SuppressWarnings("unchecked")
235                         Map<String, Object> settings = mapper.treeToValue(settingsNode, Map.class);
236                         outputWriter.setSettings(settings);
237                         if (settings.containsKey("enabled")) {
238                             outputWriter.setEnabled(Boolean.valueOf(String.valueOf(settings.get("enabled"))));
239                         }
240                     } else {
241                         logger.warn("Ignore invalid node {}", outputWriterNode);
242                     }
243                     logger.trace("Add {}", outputWriter);
244                     outputWriters.add(outputWriter);
245                 } catch (Exception e) {
246                     throw new EmbeddedJmxTransException("Exception converting settings " + outputWritersNode, e);
247                 }
248             }
249         } else {
250             logger.warn("Ignore invalid node {}", outputWritersNode);
251         }
252         return outputWriters;
253     }

255     protected void parseQueryAttributeNode(@Nonnull Query query, @Nonnull JsonNode attributeNode) {
256         if (attributeNode.isMissingNode()) {
257         } else if (attributeNode.isValueNode()) {
258             query.addAttribute(attributeNode.asText());
259         } else if (attributeNode.isObject()) {
260             List<String> keys = null;
261 
262             JsonNode keysNode = attributeNode.path("keys");
263             if (keysNode.isMissingNode()) {
264             } else if (keysNode.isArray()) {
265                 if (keys == null) {
266                     keys = new ArrayList<String>();
267                 }
268                 Iterator<JsonNode> itAttributeNode = keysNode.elements();
269                 while (itAttributeNode.hasNext()) {
270                     JsonNode keyNode = itAttributeNode.next();
271                     if (keyNode.isValueNode()) {
272                         keys.add(keyNode.asText());
273                     } else {
274                         logger.warn("Ignore invalid node {}", keyNode);
275                     }
276                 }
277             } else {
278                 logger.warn("Ignore invalid node {}", keysNode);
279             }
280 
281             JsonNode keyNode = attributeNode.path("key");
282             if (keyNode.isMissingNode()) {
283             } else if (keyNode.isValueNode()) {
284                 if (keys == null) {
285                     keys = new ArrayList<String>();
286                 }
287                 keys.add(keyNode.asText());
288             } else {
289                 logger.warn("Ignore invalid node {}", keyNode);
290             }
291 
292             String name = attributeNode.path("name").asText();
293             JsonNode resultAliasNode = attributeNode.path("resultAlias");
294             String resultAlias = resultAliasNode.isMissingNode() ? null : resultAliasNode.asText();
295             JsonNode typeNode = attributeNode.path("type");
296             String type = typeNode.isMissingNode() ? null : typeNode.asText();
297             if (keys == null) {
298                 query.addAttribute(new QueryAttribute(name, type, resultAlias));
299             } else {
300                 query.addAttribute(new QueryAttribute(name, type, resultAlias, keys));
301             }
302         } else {
303             logger.warn("Ignore invalid node {}", attributeNode);
304         }
305     }
